######源定义 及 索引定义######
#base source
source base {
type = mysql
sql_host = localhost
sql_user = qrjh
sql_pass = qrjh1000plan
sql_db = qrjh_zhb
sql_port = 3306
sql_query_pre = SET NAMES utf8
sql_query_pre = set session query_cache_type=OFF #清缓存
}

##用户主数据源##
source user_main : base
{
#建立索引之前 把ts_user中最大用户ID写入到计数器表中
sql_query_pre = replace into sph_counter (`key`,`max_id`) values ( 'zhb_user' , (select max(uid) from ts_user) )

#sql_query = SELECT uid,uname,ename,login,title,intro,trade_str,trade_id,location,search_key,identity,last_login_time,ctime FROM ts_user where identity<5 and is_audit=1 AND is_init=1 AND `uid`<=(select max_id from sph_counter where `key`='zhb_user')
sql_query = SELECT uid,uname,ename,login,title,intro,trade_str,trade_id,location,search_key,identity,last_login_time,ctime FROM ts_user where is_audit=1 AND is_init=1 AND `uid`<=(select max_id from sph_counter where `key`='zhb_user')
#sql_query第一列id需为整数
#uname、search_key...作为字符串/文本字段，被全文索引

sql_attr_uint = identity #从SQL读取到的值必须为整数
sql_attr_uint = trade_id #从SQL读取到的值必须为整数
sql_attr_timestamp = ctime #从SQL读取到的值必须为整数，作为时间属性
sql_attr_timestamp = last_login_time #从SQL读取到的值必须为整数，作为时间属性

sql_query_info_pre = SET NAMES utf8 #命令行查询时，设置正确的字符集
sql_query_info = SELECT * FROM ts_user WHERE uid=$id #命令行查询时，从数据库读取原始数据信息
}

index user_main
{
source = user_main #对应的source名称
path = /usr/local/coreseek/var/data/zhb/user_main #请修改为实际使用的绝对路径，例如：/usr/local/coreseek/var/...

mlock = 0
morphology = none
min_word_len = 1
html_strip = 0

#中文分词配置，详情请查看： http://www.coreseek.cn/products-install/coreseek_mmseg/
charset_dictpath = /usr/local/mmseg3/etc/ #BSD、Linux环境下设置，/符号结尾
#charset_dictpath = etc/ #Windows环境下设置，/符号结尾，最好给出绝对路径，例如：C:/usr/local/coreseek/etc/...

}

source user_uname_main : base
{
    #建立索引之前 把ts_user中最大用户ID写入到计数器表中
    sql_query_pre               = replace into sph_counter (`key`,`max_id`) values (  'zhb_user_uname' , (select max(uid) from ts_user) )

    sql_query               = SELECT uid,uname,trade_id,identity,last_login_time,ctime FROM ts_user where identity<5 and is_audit=1 AND is_init=1 AND `uid`<=(select max_id from sph_counter where `key`='zhb_user_uname')
                            #sql_query第一列id需为整数
                            #uname、search_key...作为字符串/文本字段，被全文索引

    sql_attr_uint            = identity                 #从SQL读取到的值必须为整数
    sql_attr_uint            = trade_id                 #从SQL读取到的值必须为整数
    sql_attr_timestamp        = ctime                   #从SQL读取到的值必须为整数，作为时间属性
    sql_attr_timestamp        = last_login_time         #从SQL读取到的值必须为整数，作为时间属性

    sql_query_info_pre      = SET NAMES utf8                                        #命令行查询时，设置正确的字符集
    sql_query_info            = SELECT * FROM ts_user WHERE uid=$id #命令行查询时，从数据库读取原始数据信息
}
